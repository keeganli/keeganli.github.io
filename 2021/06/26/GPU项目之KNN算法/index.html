GPU项目之KNN算法

发表于 2021-06-26 分类于 并行程序设计 ， GPU项目之KNN算法

项目介绍

  1. 用CUDA设计一个KNN分类算法（K近邻算法）程序。
  2. 说明：在特征空间中查找K个最相似或者距离最近的样本，然后根据K个最相似的样本对未知样本进行分类。
  3. 通过训练集和测试集给出算法的正确率。

    设计思路

  • 定义一个用作存储每个结点的结构体，包含各个维度的值，名字，以及距离当前测试点的距离的平方。

  • 输入前一百五十行作为训练集，后三十行作为测试集。

  • 遍历测试集，首先计算所有数据集距离测试集结点的距离，根据距离大小，找出距离测试集距离最近的前三个点。

  • 根据前三个点的名字来判断测试集结点的名字，并且与实际名字做比较，用作求算法正确率。

  • GPU的优化主要体现在__global__ void Get_Len(Point*, Point) 即求距离函数中，同时对所有的训练集节点求关于当前测试集节点的距离。

  • 测试集来自加州大学（自行查找）

    KNN算法知识点

    这里网络讲解的十分全面，这里我只做简单叙述

  • KNN算法就是当预测一个新的值x的时候，根据它距离最近的K个点是什么类别来判断x属于哪个类别。

  • 数据有许多标签（维度），将数据分为测试集和训练集，对每个测试数据进行测试，找到距离它最近的前K个点，这K个点出现类别最多的，即为对此数据预测的类别。

源码

串行

#include<iostream>
#include<queue>
#include<string>
#include<map>
#include<fstream>
#include<sys/time.h>
using namespace std;

const int K = 3;
const int N = 4;  //判断的维度
const int NUM = 120;  //训练集数量
const int Test_NUM = 30;  //测试集数量
string now_name;  //当前判断测试集的正确分类名字

struct Point {
    double values[N];  //各个维度的数值
    string name;   //名字
    double Len;  //距离的平方
};

//自定义根据距离利用优先队列由近到远排序
bool operator<(const Point p1, const Point p2) {
    return p1.Len > p2.Len;
}

//获取每个点的距离
void Get_Len(Point now, Point* p) {
    for (int i = 0;i < NUM;i++) {
        p[i].Len = 0;
        for (int j = 0;j < N;j++)
            p[i].Len += (now.values[j] - p[i].values[j]) * (now.values[j] - p[i].values[j]);
    }
}

//预测输入数据的分类并判断预测是否正确
bool solve(Point now, Point* p) {

    //获取距离的平方
    Get_Len(now, p);

    //使用优先队列进行排序
    priority_queue<Point> queue;
    for (int i = 0;i < NUM;i++) {
        queue.push(p[i]);
    }

    //获取距离前K近的点
    Point tmp;
    map<string, int> nameNum;  //“名字”与“数量”之间的映射
    cout << endl << "-----------------------------------" << endl;
    for (int i = 0;i < K;i++) {
        tmp = queue.top();
        queue.pop();
        nameNum[tmp.name]++;
        cout << "第" << nameNum[tmp.name] << "个" << tmp.name << ",距离的平方为：" << tmp.Len << endl;
    }

    //前K近的点中数量最多的即是预测的类型
    map<string, int>::iterator it = nameNum.begin();
    string maxName = it->first;
    for (++it;it != nameNum.end();++it) {
        if (nameNum[maxName] < it->second) {
            maxName = it->first;
        }
    }

    cout << "这是:" << maxName << endl;
    cout << endl << "-----------------------------------" << endl << endl << endl;

    //判断是否正确并返回
    if (maxName == now_name)
        return true;
    else
        return false;
}

double Get_Time(timeval start, timeval end) {

    //类定义
    // timeval
    // {
    //     time_t tv_sec;  //秒 [long int]
    //     suseconds_t tv_usec;  //微秒 [long int]
    // };

    //单位微秒
    return 1e6 * (end.tv_sec - start.tv_sec) + end.tv_usec - start.tv_usec;

}

int main() {

    //从"iris.txt"文件读入
    freopen("iris.txt", "r", stdin);

    timeval start_time, end_time;

    Point p[NUM];
    Point now;

    //读取训练集
    char trash;
    for (int i = 0;i < NUM;i++) {
        cin >> p[i].values[0] >> trash >> p[i].values[1] >> trash >> p[i].values[2] >> trash >> p[i].values[3] >> trash >> p[i].name;
    }

    gettimeofday(&start_time, 0);

    //读取数据集并解决问题
    int rightNum = 0;
    for (int i = 0;i < Test_NUM;i++) {
        cin >> now.values[0] >> trash >> now.values[1] >> trash >> now.values[2] >> trash >> now.values[3] >> trash >> now_name;
        if (solve(now, p))
            rightNum++;
    }

    gettimeofday(&end_time, 0);

    printf("正确率为：%d%\n", rightNum * 100 / Test_NUM);

    //输出在CPU的运行时间
    printf("CPU cost time = %f ms\n", Get_Time(start_time, end_time) / 1000);

    return 0;
}

并行

#include<iostream>
#include<queue>
#include<string>
#include<map>
#include<fstream>
#include<sys/time.h>
using namespace std;

const int K = 3;
const int N = 4;  //判断的维度
const int NUM = 120;  //训练集数量
const int Test_NUM=30;  //测试集数量
string now_name;  //当前判断测试集的正确分类名字

struct Point {
    double values[N];  //各个维度的数值
    string name;   //名字
    double Len;  //距离的平方
};

//自定义根据距离利用优先队列由近到远排序
bool operator<(const Point p1, const Point p2) {
    return p1.Len > p2.Len;
}

//获取每个点的距离
__global__ void Get_Len(Point* now_d, Point pd[NUM]) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;

    if (i< NUM) {
        pd[i].Len= 0;
        for (int j = 0;j < N;j++)
            pd[i].Len += (now_d->values[j] - pd[i].values[j]) * (now_d->values[j] - pd[i].values[j]);
    }       
}

//预测输入数据的分类并判断预测是否正确
bool solve(Point* now_d, Point* pd) {

    //设置进程块和进程的排布方式
    dim3 theardsPerBlock(32);
    dim3 BlocksPerGrid((NUM + theardsPerBlock.x - 1)/ theardsPerBlock.x);

    //在GPU上获取距离的平方
    Get_Len << <BlocksPerGrid, theardsPerBlock >> > (now_d, pd);
    Point *p=(Point*)malloc(NUM * sizeof(Point));
    cudaMemcpy(p, pd, NUM * sizeof(Point), cudaMemcpyDeviceToHost);
    

    //使用优先队列进行排序
    priority_queue<Point> queue;
    for (int i = 0;i < NUM;i++) {
        queue.push(p[i]);
    }
    
    //获取距离前K近的点
    Point tmp;
    map<string, int> nameNum;  //“名字”与“数量”之间的映射
    cout<<endl<<"-----------------------------------"<<endl;
for (int i = 0;i < K;i++) {

        tmp = queue.top();
        queue.pop();
        nameNum[tmp.name]++;
        cout<<"第"<<nameNum[tmp.name]<<"个"<<tmp.name<<",距离的平方为："<<tmp.Len<<endl;
    }
    

    //前K近的点中数量最多的即是预测的类型
    map<string, int>::iterator it = nameNum.begin();
    string maxName = it->first;
for (++it;it != nameNum.end();++it) {

        if (nameNum[maxName] < it->second) {
            maxName = it->first;
        }
    }

    cout << "这是:" << maxName << endl;
    cout<<endl<<"-----------------------------------"<<endl<<endl<<endl;

    //判断是否正确并返回
    if (maxName == now_name)
        return true;
    else
        return false;
}

double Get_Time(timeval start, timeval end) {

    //类定义
    // timeval
    // {
    //     time_t tv_sec;  //秒 [long int]
    //     suseconds_t tv_usec;  //微秒 [long int]
    // };

//单位微秒

    return 1e6 * (end.tv_sec - start.tv_sec) + end.tv_usec - start.tv_usec;

}

int main() {

    //从"iris.txt"文件读入
    freopen("iris.txt", "r", stdin);

    Point p[NUM];
    Point now;
    Point* pd;
    Point* now_d;

    timeval GPU_start_time, GPU_end_time;

    cudaMalloc((void**)&pd, NUM * sizeof(Point));
    cudaMalloc((void**)&now_d, sizeof(Point));

    //读取训练集
    char trash;
for (int i = 0;i < NUM;i++) {

        cin >> p[i].values[0] >> trash >> p[i].values[1] >> trash >> p[i].values[2] >> trash >> p[i].values[3] >> trash >> p[i].name;
}

    cudaMemcpy(pd, p, NUM * sizeof(Point), cudaMemcpyHostToDevice);

    gettimeofday(&GPU_start_time, 0);
    
    //读取数据集并解决问题
    int rightNum=0;
for (int i = 0;i < Test_NUM;i++) {


        cin>>now.values[0]>>trash>>now.values[1]>>trash>>now.values[2]>>trash>>now.values[3]>>trash>>now_name;
        cudaMemcpy(now_d, &now, sizeof(Point), cudaMemcpyHostToDevice);
        if(solve(now_d,pd))
            rightNum++;
    } 
    
    gettimeofday(&GPU_end_time, 0);

    printf("正确率为：%d%\n",rightNum*100/Test_NUM);

    //输出在GPU的运行时间
    printf("GPU cost time = %lf ms\n", Get_Time(GPU_start_time, GPU_end_time) / 1000);

    //释放内存
    cudaFree(pd);
    cudaFree(now_d);

    return 0;
}

性能对比分析

  • 并行程序的效率优于串行程序。
  • 同时根据算法逻辑可知，K值的选取只和正确率有关，和程序运行的效率无关。
  • 同时，对于并行程序来说，程序的运行效率和训练集数量的选取多少无关，与向量维度有关。
