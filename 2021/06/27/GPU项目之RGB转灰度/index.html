GPU项目之RGB转灰度

发表于 2021-06-27 更新于 2021-06-26 分类于 并行程序设计 ， GPU项目之RGB转灰度

项目介绍

  • 用CUDA设计一个将RGB图像转换生成灰度图像的程序。
  • 要求通过实例测试串行程序和GPU并行程序的执行效率（要求处理至少100张图片）

设计思路

  • RGB彩色图像中，一种彩色由R（红色），G（绿色），B（蓝色）三原色按比例混合而成。
  • 图像的基本单元是一个像素，一个像素需要3块表示，分别代表R，G，B，如果8为表示一个颜色，就由0-255区分不同亮度的某种原色。
  • 在我们的项目中，我们采用循环方式读入图片并且申请内存。然后让图片一张张地传到GPU中并且动态申请内存。
  • 然后使用定义的内核函数进行灰度转换，最后将结果数组拷回内存。
  • 对于图片的读入和输出，我们使用了stb_image.h库进行读入，使用stb_image_wirte.h进行输入，在读入过程中使用结构体存储每个图片的长、宽和通道数
  • 在定义内核函数trans时使用转换公式（(15da[wyn+xn+0] + 75da[wyn+xn+1] + 38da[wyn+xn+2]) >> 7;）进行像素点的灰度值变换，
  • 分别使用GPU并行方法（二维线程排布）和CPU串行方法进行灰度处理。
  • 最后使用timeval数据类型（sys_time.h)计算GPU和CPU处理图片时间。
  • 并在不同的图片规模下，比较CPU和GPU的执行效率。

设计过程

  1. 引入图形依赖库GitHub项目到服务器

    声明引用

    // 图形处理依赖库
#define STB_IMAGE_IMPLEMENTATION

  2. 内核函数处理每一张图片

    //内核函数对传入的图片并行灰度变换
__global__ void trans(unsigned char *da,unsigned char *db,int w,int h,int n)
{
    int x = blockIdx.x * blockDim.x + threadIdx.x;      //扫描的当前列址
    int y = blockIdx.y * blockDim.y + threadIdx.y;      //扫描的当前行址
    if(x < w&&y < h){                                   //小于行和高
        //灰度变换
        unsigned char ch = (15*da[w*y*n+x*n+0] + 75*da[w*y*n+x*n+1] + 38*da[w*y*n+x*n+2]) >> 7;
        db[y*w*n+x*n] = ch;
        db[y*w*n+x*n+1] = ch;
        db[y*w*n+x*n+2] = ch;
    }
}

  4. GPU灰度处理图片

    //GPU灰度处理
    unsigned char *da[101],*db[101];
    
    for(int i = 0;i < N;i++){                                                   //以一张张的方式传入GPU
                                
        //动态申请内存
        cudaMalloc((void**)&da[i],pho[i].size*sizeof(char)*pho[i].n);
        cudaMalloc((void**)&db[i],pho[i].size*sizeof(char)*pho[i].n);
        cudaMemcpy(da[i],data[i],pho[i].size*pho[i].n,cudaMemcpyHostToDevice);
        dim3 dg(32,32);                                                          //一个线程块的线程排布方式 32*32 个
        dim3 dbs((pho[i].w+dg.x-1)/dg.x,(pho[i].h+dg.y-1)/dg.y);                 //线程块的排布方式
        gettimeofday(&staG,0);                                                   //获取当前准确时间
        trans<<<dbs,dg>>>(da[i],db[i],pho[i].w,pho[i].h,pho[i].n);
        gettimeofday(&endG,0);
        Gtime += 1e6 * (endG.tv_sec - staG.tv_sec) + endG.tv_usec - staG.tv_usec; // GPU运行时间（ms）
        //将结果数组内容考回内存
        cudaMemcpy(gray2[i],db[i],pho[i].size*3,cudaMemcpyDeviceToHost);
    }
    //释放内存
    cudaFree(da);
    cudaFree(db);

  7. 完整原码

    #include <bits/stdc++.h>
#include <string>
#include <cstdio>
#include <cstdlib>
// Linux系统的日期时间头文件
#include <sys/time.h>
// 图形处理依赖库
#define STB_IMAGE_IMPLEMENTATION
#include "stb_image.h"        // 读入图片依赖库
#define STB_IMAGE_WRITE_IMPLEMENTATION
#include "stb_image_write.h"  // 输出图片依赖库
#define N 100   // 读入图片的张数
using namespace std;
/*
*存储每张图片大小等信息
*photos'informations
*
*/
struct phoz
{
    int w,h,n;//宽，高，通道数
    int size;//像素点总数
}pho[101];
/*
*读入图片时
*对路径1到100
*进行数字到字符串的转换
*/
char *change(int n)
{
    char *ans;
    ans = (char*)malloc(sizeof(char)*5);
    //char *tmp;
    int a[5],k = 0;
    while(n > 0){
        a[k++] = n % 10;
        n /= 10;
    }
    for(int i = 0;i < k;i++){
        ans[i] = a[k-i-1]+'0';
    }
    ans[k] = '\0';
    //cout<<ans;
    return ans;
}
//内核函数对传入的图片并行灰度变换
__global__ void trans(unsigned char *da,unsigned char *db,int w,int h,int n)
{
    int x = blockIdx.x * blockDim.x + threadIdx.x;      //扫描的当前列址
    int y = blockIdx.y * blockDim.y + threadIdx.y;      //扫描的当前行址
    if(x < w&&y < h){                                   //小于行和高
        //灰度变换
        unsigned char ch = (15*da[w*y*n+x*n+0] + 75*da[w*y*n+x*n+1] + 38*da[w*y*n+x*n+2]) >> 7;
        db[y*w*n+x*n] = ch;
        db[y*w*n+x*n+1] = ch;
        db[y*w*n+x*n+2] = ch;
    } 
}

int main()
{
    /*
      data[]存储原始图片信息
      gray[]存储处理后灰度图片信息
    */
    unsigned char *data[101];
    unsigned char *gray1[101],*gray2[101];
    /*
    *运行时间
    */
    timeval staG,endG;//GPU
    timeval staC,endC;//CPU
    double Gtime = 0;
    double Ctime = 0;
    //循环读入图片
    for(int i = 0;i < N;i++){
        //图片路径设置
        char path1[20] = "";
        char *u = change(i+1);
        strcpy(path1,"../source/pic (");
        strcat(path1,u);
        strcat(path1,").jpg\0");
        //读入图片，同时为对应的灰度图片申请内存
        data[i] = stbi_load(path1,&pho[i].w,&pho[i].h,&pho[i].n,0);
        gray1[i] = (unsigned char*)malloc(pho[i].w*pho[i].h*pho[i].n*sizeof(char));
        gray2[i] = (unsigned char*)malloc(pho[i].w*pho[i].h*pho[i].n*sizeof(char));      
        //获取当前读入图片的像素个数
        pho[i].size = pho[i].w*pho[i].h;
    }
    //GPU灰度处理
    unsigned char *da[101],*db[101]; 
    for(int i = 0;i < N;i++){                                                   //以一张张的方式传入GPU                           
        //动态申请内存
        cudaMalloc((void**)&da[i],pho[i].size*sizeof(char)*pho[i].n);
        cudaMalloc((void**)&db[i],pho[i].size*sizeof(char)*pho[i].n);
        cudaMemcpy(da[i],data[i],pho[i].size*pho[i].n,cudaMemcpyHostToDevice);          
        dim3 dg(32,32);                                                          //一个线程块的线程排布方式 32*32 个
        dim3 dbs((pho[i].w+dg.x-1)/dg.x,(pho[i].h+dg.y-1)/dg.y);                 //线程块的排布方式
        gettimeofday(&staG,0);                                                   //获取当前准确时间
        trans<<<dbs,dg>>>(da[i],db[i],pho[i].w,pho[i].h,pho[i].n);
        gettimeofday(&endG,0);
        Gtime += 1e6 * (endG.tv_sec - staG.tv_sec) + endG.tv_usec - staG.tv_usec; // GPU运行时间（ms）
        //将结果数组内容考回内存
        cudaMemcpy(gray2[i],db[i],pho[i].size*3,cudaMemcpyDeviceToHost);
    }
    //释放内存
    cudaFree(da);
    cudaFree(db);
    /*
    *cpu灰度处理图片
    *cpu运行时间
    */
    gettimeofday(&staC,0);
    for(int t = 0;t < N ;t++){
        int h = pho[t].h;
        int w = pho[t].w;
        int n = pho[t].n;
        for (int dx = 0; dx < h; dx++){
            for(int dy = 0;dy < w;dy++){
                unsigned char ch = (15*data[t][w*dx*n+dy*n+0] + 75*data[t][w*dx*n+dy*n+1] + 38*data[t][w*dx*n+dy*n+2]) >> 7;
                gray1[t][dx*w*n+dy*n]= ch;
                gray1[t][dx*w*n+dy*n+1] = ch;
                gray1[t][dx*w*n+dy*n+2] = ch;
            }
        }
    }
    gettimeofday(&endC,0);
    Ctime = 1e6 * (endC.tv_sec - staC.tv_sec) + endC.tv_usec - staC.tv_usec;// CPU运行时间（ms）
    /*
    *输出图片
    */
    for(int i = 0;i < N;i++){
        //CPU图片输出路径
        char path2[20] = "";
        char *u = change(i+1);
        strcpy(path2,"../Out/CPU/pic (");
        strcat(path2,u);
        strcat(path2,"_).jpg\0");
        //GPU图片输出路径
        char path3[20] = "";
        strcpy(path3,"../Out/GPU/pic (");
        strcat(path3,u);
        strcat(path3,"__).jpg\0");
        //写出图片
        int h = pho[i].h;
        int w = pho[i].w;
        int n = pho[i].n;
        stbi_write_jpg(path2, w, h, n, gray1[i], w*n);
        stbi_write_jpg(path3, w, h, n, gray2[i], w*n);     
        //释放内存
        stbi_image_free(data[i]);
        stbi_image_free(gray1[i]);
        stbi_image_free(gray2[i]);
    }
    printf("GPUCost time : %lf ms\n",Gtime/1e3);
    printf("CPUCost time : %lf ms\n",Ctime/1e3);
    //printf("CPUCost time : %lf ms\n",Ctime2*1000);
    return 0;
}

性能对比分析

  • CPU花费的时间随数据规模变化十分明显，呈现近指数的分布
  • 对于GPU而言，因为并行计算像素灰度值，同等数据规模下GPU花费的时间明显少于CPU，呈现近线性的分布。
